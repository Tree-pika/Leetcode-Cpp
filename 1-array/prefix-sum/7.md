# 数组前缀和
* 固定左端点为0，依次计算到不同的右端点的所有区间和
* 优点是每次查询一个区间的和，只需要用两个前缀和相减即可，注意端点的开闭问题
* 读取数据优化：
  ```cpp
  /*原版 */
    cin>>arr[0];
    prefix[0] = arr[0];
    for(int i=1;i<n;i++) {
        cin>>arr[i];
        prefix[i] = prefix[i-1]+arr[i];
    }
    while(cin>>a>>b){
        sum = prefix[b]-prefix[a]+arr[a];
        cout<<sum<<endl;
    }
    /*改进 */
    int presum = 0;
    for(int i=0;i<n;i++) {
        cin>>arr[i];
        presum+=arr[i];
        prefix[i] = presum;
    }
    while(cin>>a>>b){
        if(a==0)
            sum = prefix[b];
        else
            sum = prefix[b]-prefix[a-1];
        cout<<sum<<endl;
    }
    /*使用scanf/printf加快速度 */
    int presum = 0;
    for(int i=0;i<n;i++){
        scanf("%d ",&arr[i]);
        presum+=arr[i];
        prefix[i] = presum;
    }
    int sum = 0;
    int a,b;
    while(scanf("%d%d",&a,&b)==2){
        if(a==0){
            sum=prefix[b];
        }
        else{
            sum=prefix[b]-prefix[a-1];
        }
        printf("%d\n",sum);
    }
